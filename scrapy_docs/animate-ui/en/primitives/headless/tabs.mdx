## Usage

### With Highlight

```tsx
<TabGroup>
  <TabHighlight>
    <TabList>
      <TabHighlightItem index={0}>
        <Tab index={0}>Account</Tab>
      </TabHighlightItem>
      <TabHighlightItem index={1}>
        <Tab index={1}>Password</Tab>
      </TabHighlightItem>
    </TabList>
  </TabHighlight>
  <TabPanels>
    <TabPanel>Make changes to your account here.</TabPanel>
    <TabPanel>Change your password here.</TabPanel>
  </TabPanels>
</TabGroup>
```

### Without Highlight

```tsx
<TabGroup>
  <TabList>
    <Tab index={0}>Account</Tab>
    <Tab index={1}>Password</Tab>
  </TabList>
  <TabPanels>
    <TabPanel>Make changes to your account here.</TabPanel>
    <TabPanel>Change your password here.</TabPanel>
  </TabPanels>
</TabGroup>
```
